func AddNoteCategory(c *gin.Context) {
var category models.Category
err := c.ShouldBind(&category)
fmt.Printf(“%#v\n”, category)

if err != nil {
    fmt.Printf("%#v\n", err)
}

valid := validation.Validation{}
valid.Required(category.ModuleUuid, "module_uuid").Message("模块ID不能为空")
valid.Required(category.CategoryName, "category_name").Message("名称不能为空")
valid.MaxSize(category.CategoryName, 100, "category_name").Message("名称最长为100字符")
valid.Range(category.State, 0, 1, "state").Message("状态只允许0或1")

code := e.INVALID_PARAMS
if ! valid.HasErrors() {
    if ! models.ExistNoteCategoryByName(category.CategoryName) {
        code = e.SUCCESS
        models.AddNoteCategory(category.ModuleUuid, category.CategoryName, category.State)
    } else {
        code = e.ERROR_EXIST_NOTE_CATEGORY
    }
}

c.JSON(http.StatusOK, gin.H{
    "code": code,
    "msg": e.GetMsg(code),
    "data": make(map[string]string),
})

}
