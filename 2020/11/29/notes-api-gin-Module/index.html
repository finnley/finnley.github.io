//新增模块
func AddModule(c *gin.Context) {
var module models.Module
err := c.ShouldBind(&module)

code := e.INVALID_PARAMS
if err != nil {
    code = e.INVALID_PARAMS
}

valid := validation.Validation{}
valid.Required(module.ModuleName, "module_name").Message("名称不能为空")
valid.MaxSize(module.ModuleName, 100, "module_name").Message("名称最长为100字符")
valid.Range(module.State, 0, 1, "state").Message("状态只允许0或1")

if ! valid.HasErrors() {
    if ! models.ExistModuleByName(module.ModuleName) {
        code = e.SUCCESS
        models.AddModule(module.ModuleName, module.State)
    } else {
        code = e.ERROR_EXIST_MODULE
    }
}

c.JSON(http.StatusOK, gin.H{
    "code": code,
    "msg": e.GetMsg(code),
    "data": make(map[string]string),
})

}
